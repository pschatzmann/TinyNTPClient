168 bool ntpExchange(uint32_t txTm_s) {
169 NTPPacket packet = {};
170 packet.li_vn_mode = 0b11100011;
171 packet.txTm_s = txTm_s;
172 packet.txTm_f = 0;
173 _udp.begin(_port);
174
175 _udp.beginPacket(_server, _port);
176 _udp.write(reinterpret_cast<uint8_t*>(&packet), sizeof(NTPPacket));
177 _udp.endPacket();
178
179 uint32_t timeout_ms = millis() + _timeoutMs;
180 // Wait for response with timeout
181 while (!_udp.parsePacket()) {
182 if (millis() > timeout_ms) {
183 return false;
184 }
185 }
186 uint32_t t3_millis = millis();
187 NTPPacket response = {};
188 int len =
189 _udp.read(reinterpret_cast<uint8_t*>(&response), sizeof(NTPPacket));
190 if (len < (int)sizeof(NTPPacket)) {
191 return false;
192 }
193
194 bool useOffset = (txTm_s != 0);
195 if (useOffset) {
196 // Full NTP offset calculation
197 uint32_t originate = response.origTm_s;
198 uint32_t receive = response.rxTm_s;
199 uint32_t transmit = response.txTm_s;
200 uint32_t t3_unix = getTimeSec();
201 uint32_t t3_ntp = t3_unix + 2208988800UL;
202 long offset =
203 ((long)(receive - originate) + (long)(transmit - t3_ntp)) / 2;
204 _lastNtpTime = transmit - 2208988800UL + offset;
205 _lastUpdateMillis = t3_millis;
206 } else {
207 // Only use server transmit timestamp
208 uint32_t transmit = response.txTm_s;
209 _lastNtpTime = transmit - 2208988800UL;
210 _lastUpdateMillis = t3_millis;
211 }
212 return true;
213 }
