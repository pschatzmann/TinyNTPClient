205 bool ntpExchange(uint32_t txTm_s) {
206 NTPPacket packet = {};
207 // Set LI=3 (no warning), VN=3 (NTPv3), Mode=3 (client)
208 // Binary: 11 011 011 = 0xDB
209 packet.li_vn_mode = 0xDB;
210 // Convert transmit timestamp to network byte order
211 packet.txTm_s = l_htonl(txTm_s);
212 packet.txTm_f = 0;
213
214 _udp.begin(_port);
215 _udp.beginPacket(_server, _port);
216 _udp.write(reinterpret_cast<uint8_t*>(&packet), sizeof(NTPPacket));
217 _udp.endPacket();
218
219 // Wait for response with timeout
220 uint32_t timeout_ms = ::millis() + _timeoutMs;
221 int packetSize = 0;
222 while ((packetSize = _udp.parsePacket()) == 0) {
223 if (::millis() > timeout_ms) {
224 log("NTP: request timed out\n");
225 return false;
226 }
227 }
228
229 // Read response
230 uint32_t t3_millis = ::millis();
231 NTPPacket response = {};
232 uint8_t* buffer = reinterpret_cast<uint8_t*>(&response);
233
234 // Check if packet size is correct
235 if (packetSize < (int)sizeof(NTPPacket)) {
236 _udp.read(buffer, packetSize); // Read available data for logging
237 log("NTP: packet too short - Expected: %d, Got: %d (%s)\n",
238 sizeof(NTPPacket), packetSize, buffer);
239 return false;
240 }
241
242 // Read the packet data - read in chunks if necessary
243 int totalRead = 0;
244 int bytesToRead = sizeof(NTPPacket);
245
246 while (totalRead < bytesToRead && _udp.available()) {
247 int bytesRead = _udp.read(buffer + totalRead, bytesToRead - totalRead);
248 if (bytesRead > 0) {
249 totalRead += bytesRead;
250 } else {
251 break;
252 }
253 }
254
255 if (totalRead < (int)sizeof(NTPPacket)) {
256 log("NTP: response read incomplete - Expected: %d, Got: %d (%s)\n",
257 sizeof(NTPPacket), totalRead, buffer);
258 return false;
259 }
260
261 bool useOffset = (txTm_s != 0);
262 if (useOffset) {
263 // Full NTP offset calculation
264 uint32_t originate = l_ntohl(response.origTm_s);
265 uint32_t receive = l_ntohl(response.rxTm_s);
266 uint32_t transmit = l_ntohl(response.txTm_s);
267 // log("NTP: originate=%u, receive=%u, transmit=%u\n", originate, receive,
268 // transmit);
269 uint32_t t3_unix = getTimeSec();
270 uint32_t t3_ntp = t3_unix + 2208988800UL;
271 long offset =
272 ((long)(receive - originate) + (long)(transmit - t3_ntp)) / 2;
273 _lastNtpTime = transmit - 2208988800UL + offset;
274 _lastUpdateMillis = t3_millis;
275 // log("NTP: _lastNtpTime=%u, _lastUpdateMillis=%u\n", _lastNtpTime,
276 // _lastUpdateMillis);
277 } else {
278 // Only use server transmit timestamp
279 uint32_t transmit = l_ntohl(response.txTm_s);
280 // log("NTP: raw transmit=%u (0x%08X), converted=%u\n", response.txTm_s,
281 // response.txTm_s, transmit);
282 _lastNtpTime = transmit - 2208988800UL;
283 _lastUpdateMillis = t3_millis;
284 // log("NTP: _lastNtpTime=%u, _lastUpdateMillis=%u, t3_millis=%u\n",
285 // _lastNtpTime, _lastUpdateMillis, t3_millis);
286 }
287 return true;
288 }
